从零开始使用Vue3.0和Vant制作音乐App (一)1️⃣

· 2020-08-28 · # JavaScript # Vue.js # HTML # CSS

模仿网易云制作音乐APP🎧

开始之前

  • 网易云API 这个API可以通过node.js实现抓取网易云API
  • vant 一个简单好用的vue组件库

项目创建

vue create vue-music-project

  • 安装Vue-router

     npm install vue-router -s

    • 在src下创建router文件夹创建index.js 这里主要处理路由的跳转

    • 新建时写入如下

    • import Vue from 'vue'
import VueRouter from 'vue-router'
//安装router
Vue.use(VueRouter)

//组件
const Home = () => import ('../views/home/home')

//路由跳转
const routes = [
  {
    path: '',
    redirect: '/home',
  },
  {
    path: '/home',
    component: Home,
    meta: {
      keepAlive: true
    }
  }
]

//路由
const router = new VueRouter({
  routes,
  mode: 'history'
})

//解决有时候路由跳转报错
const originalPush = VueRouter.prototype.push
VueRouter.prototype.push = function push(location) {
  return originalPush.call(this, location).catch(err => err)
}

//导出路由对象
export default router

  • 安装Vuex

    npm i vuex -s

    • 在src下创建store文件夹下创建index.js 这里处理vuex全局效果

    • 创建时写入代码如下

    • import Vue from 'vue'
import Vuex from 'vuex'

//安装vuex
Vue.use(Vuex)

//要使用的对象
const state = {
  name: '',
}

const store = new Vuex.Store({
  state
})

export default store

  • 安装axios

    npm install axios --save

    • 在src下新建network文件夹下新建request.js 这里主要处理网络请求

    • 创建时写入如下代码

    • import axios from 'axios'
import baseUrl from "这里放要请求的总路径";

export function request(config,success, failure){
  const instance = axios.create({
    // baseURL:'https://autumnfish.cn',
    baseURL:baseUrl,
    timeout:5000
  })
  
  // 2. axios的拦截器
  //2.1 请求拦截
  //1. 比如说 config中的一些信息不符合服务器的要求
  //2. 每次发送网络请求时, 都希望在界面中添加一个请求的图标
  //3. 某些网络请求(比如 登录token) 必须携带一些特殊的信息
  instance.interceptors.request.use(config => {
    return config
  }, err => {

  });

  // 2.2响应拦截
  instance.interceptors.response.use(res=>{
    //响应拦截必须返回res.data
    return res.data
  },err=>{
    console.log(err);
  });
  
  //返回promise
  return instance(config)
}

  • main.js的配置

    •   import Vue from 'vue'
  import App from './App.vue'
  //路由
  import router from "./router"
  //vuex
  import store from "./store";
  
  //使用vant
  import Vant from 'vant';
  import 'vant/lib/index.css';
  Vue.use(Vant);
  
  
  //将vuex的store添加到Vue的原型中 , 即所有组件中
  Vue.prototype.$store = store;
  
  
  import axios from 'axios'
  axios.defaults.withCredentials = true
  Vue.prototype.$axios = axios
  
  
  new Vue({
    render: h => h(App),
    router
  }).$mount('#app')
