v-show

控制元素的显示状态 只是操作样式

	<div id="app">
        <input type="button" value="隐藏或显示" @click="clickShow">
        <input type="button" value="age++++++" @click="addAge">
        <h1 v-show="isShow" style="background: red; color: white;">Hello world</h1>
        <h1 v-show="age<=18" style="background: red; color: white;">Hello world!!!!!!!!!!!!!</h1>
        
    </div>
    <script src="js/vue.js"></script>
    <script>
        let app = new Vue({
            el:"#app",
            data:{
                isShow:false,
                age:18
            },
            methods:{
                clickShow:function(){
                    this.isShow = !this.isShow;
                },
                addAge:function(){
                    this.age++;
                }
            }
        })

    </script>

            </h2>

        let app = new Vue({
            el: "#app",

                    { name: "西红柿炒鸡蛋" },

计算属性

computed:可以实现计算

let app = new Vue({
    el:"#app",
    data:{
        math:80,
        english:20,
        physics:50
    },
    methods:{

    },
    computed:{
        sum: function () {
            return this.math+this.english+this.physics;
        },
        avg: function () {
            return Math.round(this.sum/3);
        }
    }
})

*Vue组件

组件是可复用的 Vue 实例，且带有一个名字：在这个例子中是 我们可以在一个通过 new Vue 创建的 Vue 根实例中，把这个组件作为自定义元素来使用：

//新建一个组件名为button-counter，data必须用函数来返回
Vue.component('button-counter', {
  data: function () {
    return {
      count: 0
    }
  },
    //组件写在这个属性里
  template: '<button v-on:click="count++">You clicked me {{ count }} times.</button>'
});
//用一个vue实例才能调用
new Vue({ el: '#components-demo' });

然后我们就可以在html中直接写<button-counter></button-counter>标签即可，当然标签需要写在一个vue的实例中

但是这样做他会在每个vue实例中都会显示，我们也可以把它绑定在一个vue实例中

/*
这里只会显示一个按钮
<div id="app">
    <alert></alert>
</div>
app2中不显示内容
<div id="app2">
    <alert></alert>
    <alert></alert>
</div>
*/
//把控件赋值给一个对象
let Alert = {template:'<button @click="on_click">弹弹弹</button>',
                methods: {
                    on_click: function () {
                        alert('hello world');
                    }
                }
            };
//然后在vue实例中用componets属性把
new Vue({
    el: '#app',
    components:{
        alert:Alert,
    }
});

这样他就只会在id为app的div中显示，其他div中是不显示的

把组件写在js中总是麻烦，我们也可以把它写在html中

<script>
    //新建一个叫like的组件
Vue.component('like',{
    //把在HTML中一个template标签中并且id为#like-cheked里的html内容作为组件内容
    template: '#like-cheked',
    data:function () {
        return{
            like_count:10,
            liked:false,
        }
    },
    //这里是他的方法
    methods: {
        toggle_like:function () {
            if(!this.liked){
                this.like_count++;
            }else{
                this.like_count--;
            }
            this.liked = !this.liked;
        }
    }
});
//一个实例
new Vue({
    el:"#app3",
});
//当然也可以直接绑定摘app3中
</script>

        data:function () {

    //这是一个子级的组件
    Vue.component('show',{
        template: `
			<button @click="on_click()">显示余额</button>
		`,
        methods: {
            on_click(){
                // 传递参数
                this.$emit('show-balance',{a: 1, b: 2});
            }
        }
    });
